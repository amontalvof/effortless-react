# Overview

`useMediaQuery` is a hook that allows you to write media queries that are going to render out javascript code so you can do several things depending on the result of the media query.

### useMediaQuery

```ts
const useMediaQuery = (mediaQuery: string) => {
    const [isMatch, setIsMatch] = useState(false);
    const [mediaQueryList, setMediaQueryList] =
        useState<MediaQueryList | null>(null);

    useEffect(() => {
        const list = window.matchMedia(mediaQuery);
        setMediaQueryList(list);
        setIsMatch(list.matches);
    }, [mediaQuery]);

    useEventListener(
        'change',
        (event: any) => setIsMatch(event.matches),
        mediaQueryList
    );

    return isMatch;
};
```

### Usage

```tsx
const MediaQueryComponent = () => {
    const isLarge = useMediaQuery('(min-width: 700px)');

    return (
        <StyledContainer>
            <h1>Large: {isLarge.toString()}</h1>
        </StyledContainer>
    );
};
```
